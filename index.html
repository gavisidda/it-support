    }

      allow read: if request.auth != null;
      allow write: if request.auth != null &&

    }
    match /requests/{rid} {
      allow read: if request.auth != null;
      allow create: if request.auth != null;
      allow update: if request.auth != null &&
        (isAdmin() || request.auth.uid == resource.data.requestorId
        || request.auth.uid == resource.data.professionalId
        || (resource.data.status == 'pending' &&
          resource.data.professionalId == null));
      allow delete: if isAdmin();
      match /messages/{mid} {
        allow read, write: if request.auth != null;
      }
    }
    match /platform/{doc} {
      allow read: if request.auth != null;
      allow write: if isAdmin();
    }
    match /broadcasts/{bid} {
      allow read: if request.auth != null;
      allow write: if isAdmin();
    }
  }
}

⚙️ Firebase Setup Guide

Copy these rules to fix Google Auth and Chat permissions

🔴 Step 1 — Fix Firestore Security Rules

Go to Firebase Console → Firestore → Rules and replace with:

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    function isAdmin() {
      return request.auth != null &&
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin';
    }

    // Platform settings — any logged-in user can read; only admin writes
    match /platform/{doc} {
      allow read: if request.auth != null;
      allow write: if isAdmin();
    }

    // User profiles
    match /users/{userId} {
      allow read: if request.auth != null;
      allow write: if request.auth != null &&
        (request.auth.uid == userId || isAdmin());
    }

    // Service Requests
    match /requests/{requestId} {
      allow read: if request.auth != null;
      allow create: if request.auth != null;
      allow update: if request.auth != null && (
        isAdmin() ||
        request.auth.uid == resource.data.requestorId ||
        request.auth.uid == resource.data.professionalId ||
        (resource.data.status == 'pending' &&
         resource.data.professionalId == null &&
         request.resource.data.professionalId == request.auth.uid)
      );
      allow delete: if isAdmin();

      // ✅ MESSAGES SUBCOLLECTION — this fixes the chat permission error
      match /messages/{msgId} {
        allow read, write: if request.auth != null;
      }
    }
  }
}
🟡 Step 2 — Enable Google Sign-In
  1. Go to Firebase Console → Authentication → Sign-in method
  2. Click Google → toggle Enable → choose your project support email → Save
  3. Go to Authentication → Settings → Authorized domains
  4. Add your-username.github.io (or your custom domain)
🔵 Step 3 — Create First Admin
  1. Register a normal account through the app
  2. Go to Firebase Console → Firestore → users collection
  3. Find your user document → change role from "requestor" to "admin"
  4. Log out and back in — you'll see the Admin Dashboard
  5. After that, promote others via Admin → Users → Make Admin
🟢 Required Firestore Indexes

Create these in Firestore → Indexes → Composite:

requests | requestorId ▲ | createdAt ▼
requests | status ▲ | createdAt ▼
requests | professionalId ▲ | createdAt ▼

💡 Tip: Firebase auto-suggests missing indexes when queries first run — just click the link in your browser console.
